# Quaternion multiplication
def quat_mult(q1, q2):
    w0, x0, y0, z0 = q1
    w1, x1, y1, z1 = q2
    return np.array([
        w0*w1 - x0*x1 - y0*y1 - z0*z1,
        w0*x1 + x0*w1 + y0*z1 - z0*y1,
        w0*y1 - x0*z1 + y0*w1 + z0*x1,
        w0*z1 + x0*y1 - y0*x1 + z0*w1
    ])

# Quaternion kinematics
def dqdt(q, w):
    w_quat = np.concatenate(([0], w))  # Use parentheses to keep 1D
    return 0.5 * quat_mult(q, w_quat)



# B-dot control torque
def bdot_torque(w, B):
    dB = -np.cross(w, B)
    m = -k_B * dB
    return np.cross(m, B)

    q1, q2, q3, q4 = q  # q = [w, x, y, z]

    return np.array([
        [q1**2 + q2**2 - q3**2 - q4**2, 2*(q2*q3 + q1*q4),       2*(q2*q4 - q1*q3)],
        [2*(q2*q3 - q1*q4),            q1**2 - q2**2 + q3**2 - q4**2, 2*(q3*q4 + q1*q2)],
        [2*(q2*q4 + q1*q3),            2*(q3*q4 - q1*q2),       q1**2 - q2**2 - q3**2 + q4**2]
    ])

def aero_torque(w, q):
    # Declare a function attribute to track first call
    if not hasattr(aero_torque, "has_printed"):
        aero_torque.has_printed = False

    n_B = np.array([0, 0, 1])  # Body frame z-axis

    A_q = q_to_a(q)

    # Dummy position for Earth rotation correction
    x, y = 5, 5
    xd, yd, zd = 10, 10, 10

    # Compute relative velocity in inertial frame
    v_rel_I = np.array([xd + w_E * y, yd - w_E * x, zd])

    # Transform to body frame
    v_B = A_q @ v_rel_I

    if norm(v_B) == 0:
        cos_theta = 0
    else:
        cos_theta = np.dot(v_B, n_B) / norm(v_B)
    cos_theta = np.clip(cos_theta, 0, 1)

    rho = compute_density(altitude)
    v_rel_mag = norm(v_B)
    F_drag = 0.5 * rho * v_rel_mag**2 * Cd * A
    F_vector = -F_drag * n_B
    T_aero = np.cross(r_cp, F_vector)

    # Only print once
    if not aero_torque.has_printed:
        print(f"--- Aero Torque (Single Print) ---")
        print(f"Altitude: {altitude / 1e3:.0f} km")
        print(f" (density): {rho:.3e} kg/m")
        print(f"F_drag magnitude: {F_drag:.3e} N")
        print(f"Torque: {T_aero} Nm")
        print("----------------------------------")
        aero_torque.has_printed = True

    return T_aero

# Dynamics function
def dynamics(t, state):
    q = state[0:4]
    w = state[4:7]
    B = magnetic_field(t)
    T_bdot = bdot_torque(w, B)
    T_aero = aero_torque(w, q)
    total_torque = T_bdot + T_aero
    dw = I_inv @ (total_torque - np.cross(w, I @ w))
    dq = dqdt(q, w)
    return np.concatenate([dq, dw])

# RK4 Integrator
def rk4(f, t, y, dt):
    k1 = f(t, y)
    k2 = f(t + dt/2, y + dt/2 * k1)
    k3 = f(t + dt/2, y + dt/2 * k2)
    k4 = f(t + dt, y + dt * k3)
    return y + dt/6 * (k1 + 2*k2 + 2*k3 + k4)

def q_to_a(q):
    w, x, y, z = q
    return np.array([
        [1-2*(y**2+z**2), 2*(x*y - z*w),   2*(x*z + y*w)],
        [2*(x*y + z*w),   1-2*(x**2+z**2), 2*(y*z - x*w)],
        [2*(x*z - y*w),   2*(y*z + x*w),   1-2*(x**2 + y**2)]
    ])

# Plot cube
def plot_linear_cube(ax, x_I, color='red'):
    x, y, z = x_I[0,:], x_I[1,:], x_I[2,:]
    index = [[0, 1, 2, 3], [4, 5, 6, 7], [0, 3, 7, 4], 
             [1, 2, 6, 5], [0, 1, 5, 4], [2, 3, 7, 6]]
    kwargs = {'alpha': 1, 'color': color}
    for i in index:
        ax.plot3D([x[i[0]], x[i[1]], x[i[2]], x[i[3]], x[i[0]]], 
                  [y[i[0]], y[i[1]], y[i[2]], y[i[3]], y[i[0]]],
                  [z[i[0]], z[i[1]], z[i[2]], z[i[3]], z[i[0]]], **kwargs)

# Animation function
def animate(i, *fargs):
    ax, x_B, q_save, w_save = fargs
    ax.clear()
    ax.set_xlim3d(-2,2)
    ax.set_ylim3d(-2,2)
    ax.set_zlim3d(-2,2)
    ax.set_xlabel('$X$')
    ax.set_ylabel('$Y$')
    ax.set_zlabel('$Z$')

    q = q_save[i]
    A_q = q_to_a(q)
    x_I_l = A_q.T @ x_B
    plot_linear_cube(ax, x_I_l, 'blue')

    w_B = w_save[i]
    w_I = A_q.T @ w_B
    ax.quiver(0, 0, 0, *w_I, length=3, normalize=True, color='r')
    scale_w = 3 / norm(w_I)
    ax.text(*(scale_w*w_I), 'w', color='red', fontsize=12)

    z_B = np.array([0, 0, 1])
    z_I = A_q.T @ z_B
    ax.quiver(0, 0, 0, *z_I, length=3, normalize=True, color='g')
    ax.text(*(3*z_I), 'e3', color='g', fontsize=12)

    H_B = J_B @ w_B
    H_I = A_q.T @ H_B
    ax.quiver(0, 0, 0, *H_I, length=5, normalize=True, color='b')

    theta = np.degrees(acos(np.dot(H_I / norm(H_I), z_I / norm(z_I))))
    beta = np.degrees(acos(np.dot(w_I / norm(w_I), z_I / norm(z_I))))
    ax.text(*(4*z_I), f'theta={theta:.2f}\nbeta={beta:.2f}', color='k', fontsize=12)
